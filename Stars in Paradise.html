234  def takeDamage(character):
235      #Handle character taking damage
236      global kaelHealth, lumaHealth, kaelCollisionCooldown, lumaCollisionCooldown
237  
238      if character == "kael" and kaelCollisionCooldown <= 0:
239          kaelHealth = max(0, kaelHealth - 1)
240          kaelCollisionCooldown = COLLISION_COOLDOWN_TIME
241  
242      elif character == "luma" and lumaCollisionCooldown <= 0:
243          lumaHealth = max(0, lumaHealth - 1)
244          lumaCollisionCooldown = COLLISION_COOLDOWN_TIME
245          """we use the max() function here instead of -= because this would trigger game over. so once the  
246           character health is -ve, it automatically sets to 0. Also we use the cool down to basically  
247           stop the health from decreasing every frame the enemy is colliding with the characters"""

298  def spawnObstacle():
299      # Spawn new obstacles as we scroll, randomly
300      global nextObstacleDistance, worldDistance
301  
302      if worldDistance >= nextObstacleDistance:
303          # Randomly choose enemy type (0, 1, or 2 for enemy1, enemy2, enemy3)
304          enemyType = random.randint(0, 2)
305  
306          # Create new enemy obstacle
307          obstacle = {
308              'x': WIDTH + random.randint(0, 200),
309              'y': random.randint(300, 550),
310              'width': 70,  # Fixed width based on your enemy sprite size
311              'height': 80,  # Fixed height based on your enemy sprite size
312              'enemyType': enemyType,  # 0=enemy1, 1=enemy2, 2=enemy3
313              'currentFrame': 0.0,  # Current animation frame (float for smooth animation)
314              'color': (random.randint(100, 255), random.randint(100, 255), random.randint(100, 255)),  # Keep as fallback
315              'isHit': False,  # Track if enemy has been hit
316              'hitTime': 0  # When the enemy was hit
317          }
318  
319          obstacles.append(obstacle)
320          nextObstacleDistance += random.randint(300, 800)

323  def updateObstacles(scrollSpeed):
324      #Move obstacles with the scrolling and remove off-screen ones
325      global obstacles
326  
327      for obstacle in obstacles[:]:
328          obstacle['x'] -= scrollSpeed
329  
330          # Update enemy animation only if not hit
331          if not obstacle['isHit']:
332              obstacle['currentFrame'] += enemyAnimationSpeed
333  
334              # Reset animation frame based on enemy type (all have 2 frames)
335              if obstacle['currentFrame'] >= 2:
336                  obstacle['currentFrame'] = 0.0
337  
338          # Remove obstacles that have scrolled off the left side
339          if obstacle['x'] + obstacle['width'] < -100:
340              obstacles.remove(obstacle)
341  
342  
343  def drawObstacles():
344      #Draw all current obstacles with enemy sprites or stars if hit
345      current_time = pygame.time.get_ticks()
346  
347      for obstacle in obstacles[:]:
348          if obstacle['isHit']:
349              # Draw star instead of enemy
350              screen.blit(star, (obstacle['x'], obstacle['y']))
351  
352              # Remove star after 1 second (1000 milliseconds)
353              if current_time - obstacle['hitTime'] > 1000:
354                  obstacles.remove(obstacle)
355          else:
356              # Get the appropriate enemy sprite list and current frame
357              enemyType = obstacle['enemyType']
358              frameIndex = int(obstacle['currentFrame'])
359  
360              if enemyType == 0:
361                  # Use enemy1 sprites
362                  screen.blit(enemy1[frameIndex], (obstacle['x'], obstacle['y']))
363              elif enemyType == 1:
364                  # Use enemy2 sprites
365                  screen.blit(enemy2[frameIndex], (obstacle['x'], obstacle['y']))
366              elif enemyType == 2:
367                  # Use enemy3 sprites
368                  screen.blit(enemy3[frameIndex], (obstacle['x'], obstacle['y']))
369              else:
370                  # Fallback to rectangle if something goes wrong
371                  pygame.draw.rect(screen, obstacle['color'],
372                                   (obstacle['x'], obstacle['y'], obstacle['width'], obstacle['height']))

374  def obstacleCollision():
375      # Handle all collision detection between characters and enemies
376      global kaelCollisionCooldown, lumaCollisionCooldown
377  
378      # Update collision cool downs, bringing back to 0 since if cool down is too large
379      # the damage happens less often in fps, even if collision occurs,
380      # CHECK CONDITION IN TAKE DAMAGE
381      if kaelCollisionCooldown > 0:
382          kaelCollisionCooldown -= 1
383      if lumaCollisionCooldown > 0:
384          lumaCollisionCooldown -= 1
385  
386      kaelRect = getCharacterRect(kaelX, kaelY)
387      lumaRect = getCharacterRect(lumaX, lumaY)
388  
389      for obstacle in obstacles[:]:
390          if obstacle['isHit']: # "if obstacle isHit == True"
391              continue  # Skip hit enemies
392  
393          enemyRect = getEnemyRect(obstacle)
394  
395          # Check if Kael is hitting (M key pressed) and colliding with enemy
396          if hit and checkCollision(kaelRect, enemyRect):
397              hitEnemy(obstacle)
398              continue  # Don't check for damage collision if hitting
399  
400          # Check for damage collisions (only if not in cooldown frames)
401          if not hit and checkCollision(kaelRect, enemyRect):
402              takeDamage("kael")
403  
404          if checkCollision(lumaRect, enemyRect):
405              takeDamage("luma")

421  def scrollWorld(direction, speed):
422      # Handle infinite world scrolling
423      global bgOffset, worldDistance, cameraX
424  
425      # changing the coordinates and the position values to -ve OR
426      # +ve by speed depending on the direction we're moving
427      if direction == "right":
428          bgOffset += speed
429          worldDistance += speed
430          cameraX += speed
431          updateObstacles(speed)
432          spawnObstacle()
433      elif direction == "left":
434          bgOffset -= speed
435          worldDistance -= speed
436          cameraX -= speed
437          updateObstacles(-speed)

545  def moveRightKael():
546      global currentFrameKael, kaelX, lumaX
547  
548      """the instantaneous frame being displayed has to get  
549      changed so that it can go through each index in the list of  
550      movement to the respective character  """
551      currentFrameKael += frameSpeed
552      # when the index reaches out of the list, you reset to loop the movement
553      if currentFrameKael >= len(kaelRun):
554          currentFrameKael = 0
555  
556      # Move character, but check if we should scroll instead, by checking if Kael is near the
557      # either edge of the screen.
558      if kaelX < SCROLL_RIGHT_BOUNDARY:
559          kaelX += 5
560      else:
561          # Character reached scroll boundary - scroll world instead
562          scrollWorld("right", 5)
563          lumaX -= 5
564      screen.blit(kaelRun[int(currentFrameKael)], (kaelX, kaelY))

566  def moveLeftKael():
567      global currentFrameKael, kaelX, lumaX
568  
569      # same logic but with flipped images for left
570      currentFrameKael += frameSpeed
571      if currentFrameKael >= len(kaelRun):
572          currentFrameKael = 0
573  
574      # Move character, but check if we should scroll instead
575      if kaelX > SCROLL_LEFT_BOUNDARY:
576          kaelX -= 5
577      else:
578          # Character reached scroll boundary - scroll world instead
579          scrollWorld("left", 5)
580          lumaX += 5
581      # using transform.flip function, with boolean tuples, for horizontal flip, hence True
582      screen.blit(pygame.transform.flip(kaelRun[int(currentFrameKael)], True, False), (kaelX, kaelY))

584  def jumpRightKael(gravity, ground):
585      global currentFrameKael, kaelX, kaelY, jump, velocityKael, lumaX
586  
587      currentFrameKael += frameSpeed
588      if currentFrameKael >= len(kaelJump):
589          currentFrameKael = 0
590      # the velocity is -15, so really you're subtracting 15 from the y-value
591      kaelY += velocityKael
592      # while doing so, we also want to increase the velocity each time so,
593      # we are subtracting less and less each time, bringing kael back to the ground
594      velocityKael += gravity
595  
596      # Handle horizontal movement during jump
597      if kaelX < SCROLL_RIGHT_BOUNDARY:
598          kaelX += 3
599      else:
600          scrollWorld("right", 3)
601          lumaX -= 3
602  
603      """ when the velocity becomes a positive int, and starts  
604      bring kael down from GROUND LEVEL, we set kael's y back to the ground, reset jump  
605      flag and velocity becomes -15 again to bring kael up when jump is true again"""
606      if kaelY > ground:
607          kaelY = ground
608          jump = False
